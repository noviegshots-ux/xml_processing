// Load XML from URL with CORS support function loadXMLFromURL() { const url = document.getElementById('xmlUrlInput').value.trim(); const customName = document.getElementById('xmlUrlName').value.trim(); const useProxy = document.getElementById('useProxy').checked; if (!url) { alert('Please enter a valid URL'); return; } // Validate URL try { new URL(url); } catch { alert('Please enter a valid URL'); return; } document.getElementById('status').innerHTML = 'Loading from URL...'; document.getElementById('corsInfo').classList.remove('show'); // Try direct fetch first attemptFetch(url, customName, 0, useProxy); } // Attempt to fetch with fallback to CORS proxy function attemptFetch(url, customName, proxyIndex, useProxy) { let fetchUrl = url; if (useProxy && proxyIndex < CORS_PROXIES.length) { fetchUrl = CORS_PROXIES[proxyIndex] + encodeURIComponent(url); } fetch(fetchUrl) .then(response => { if (!response.ok) throw new Error('Network error: ' + response.status); return response.text(); }) .then(xmlText => { const parser = new DOMParser(); xmlDoc = parser.parseFromString(xmlText, 'text/xml'); if (xmlDoc.getElementsByTagName('parsererror').length > 0) { throw new Error('Invalid XML'); } const fileName = customName || url.split('/').pop() || 'url_file_' + Date.now(); cacheXMLData(fileName, xmlDoc); document.getElementById('xmlUrlInput').value = ''; document.getElementById('xmlUrlName').value = ''; document.getElementById('useProxy').checked = false; document.getElementById('status').innerHTML = `✓ Loaded from URL: ${fileName}`; // Automatically display the loaded data currentPage = 1; viewAttributes(1); openTab(null, 'Tab-ViewRemove'); }) .catch(error => { // If direct fetch failed and proxy not yet tried, show info and try proxy if (!useProxy && proxyIndex === 0) { document.getElementById('corsInfo').classList.add('show'); document.getElementById('status').innerHTML = `⚠ CORS blocked. Try enabling proxy.`; } else if (useProxy && proxyIndex < CORS_PROXIES.length - 1) { // Try next proxy attemptFetch(url, customName, proxyIndex + 1, useProxy); } else { document.getElementById('status').innerHTML = `✗ Error: ${error.message}`; alert('Error loading XML from URL:\n' + error.message + '\n\nTry checking the "Use CORS Proxy" option.'); } }); }
